4) JS: autoplay diagnostics + toggle + reduced-motion
Create/patch app.js:
(function(){
  const v = document.getElementById('heroVideo');
  const b = document.getElementById('toggleVideo');
  if(!v) return;

  // Log errors to catch codec/path issues
  v.addEventListener('error', () => console.warn('Hero video error:', v.error));
  v.addEventListener('stalled', () => console.warn('Hero video stalled'));
  v.addEventListener('waiting', () => console.log('Hero video waiting/buffering'));
  v.addEventListener('loadeddata', () => console.log('Hero video loadeddata OK'));

  // Force inline autoplay on iOS
  v.muted = true;
  v.playsInline = true;

  // Reduced motion
  const m = matchMedia('(prefers-reduced-motion: reduce)');
  const applyRM = () => { if(m.matches){ v.pause(); v.removeAttribute('autoplay'); if(b) b.textContent='Play Background'; } };
  applyRM(); m.addEventListener('change', applyRM);

  if(b){
    b.addEventListener('click', () => {
      if(v.paused){ v.play(); b.setAttribute('aria-pressed','false'); b.textContent='Pause Background'; }
      else { v.pause(); b.setAttribute('aria-pressed','true'); b.textContent='Play Background'; }
    });
  }
})();
