3) CSS — transparent at top, black on scroll, cross-fade logos
Append to your main stylesheet (do not remove existing rules):
/* Header background states */
.site-header {
  background: transparent;
  transition: background .25s ease, box-shadow .25s ease, border-color .25s ease;
}
.scrolled .site-header {
  background: rgba(0,0,0,.90);
  backdrop-filter: saturate(140%) blur(8px);
  box-shadow: 0 6px 18px rgba(0,0,0,.28);
  border-bottom: 1px solid rgba(255,255,255,.06);
}

/* Logo stacking: identical position, smooth cross-fade */
.brand { position: relative; display: block; height: 32px; }
.brand .logo {
  position: absolute; inset: 0;
  height: 32px; width: auto; object-fit: contain;
  display: block;
  transition: opacity .25s ease;
}
.logo--primary  { opacity: 1; }
.logo--scrolled { opacity: 0; }

/* Swap on scroll */
.scrolled .logo--primary  { opacity: 0; }
.scrolled .logo--scrolled { opacity: 1; }

/* Reduced motion: no fades, just swap */
@media (prefers-reduced-motion: reduce) {
  .brand .logo { transition: none; }
}

4) JS — add/remove the “scrolled” class (lightweight, safe)
Append to public/app.js (or your main JS file). If you already have a scroll handler, reuse it and just ensure it toggles the scrolled class on <html>.
// Header scroll toggle: adds 'scrolled' to <html> when past top
(function(){
  const root = document.documentElement;
  const header = document.querySelector('.site-header');
  if(!header) return;

  const apply = () => {
    if (window.scrollY > 8) root.classList.add('scrolled');
    else root.classList.remove('scrolled');
  };

  apply();
  window.addEventListener('scroll', apply, { passive: true });
})();
