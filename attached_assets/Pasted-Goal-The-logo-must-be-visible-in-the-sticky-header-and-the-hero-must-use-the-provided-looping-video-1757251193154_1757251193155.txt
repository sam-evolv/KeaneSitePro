4) JS: reduced motion + pause toggle
Insert in /public/app.js (or create if missing):
(function(){
  const video = document.getElementById('heroVideo');
  const btn = document.getElementById('toggleVideo');
  if(!video || !btn) return;

  // Respect reduced motion
  const prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)');
  const handleReduced = () => { if(prefersReduced.matches){ video.pause(); video.removeAttribute('autoplay'); } };
  handleReduced(); prefersReduced.addEventListener('change', handleReduced);

  // Toggle play/pause
  btn.addEventListener('click', () => {
    if(video.paused){ video.play(); btn.setAttribute('aria-pressed','false'); btn.textContent='Pause Background'; }
    else{ video.pause(); btn.setAttribute('aria-pressed','true'); btn.textContent='Play Background'; }
  });
})();
