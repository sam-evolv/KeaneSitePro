3) Stop deep-link scrolling to the bottom
Three common causes: preserved scroll position, autofocus on a lower field, or hashes in links.
HTML checks
Ensure service links have no hash:
<a class="card__more" href="services/waste-removal.html">Learn more</a>
Remove any autofocus attributes from inputs/textarea on service pages.
JS (add to public/app.js):
// Always land at the top on navigation
document.addEventListener('DOMContentLoaded', () => {
  if ('scrollRestoration' in history) history.scrollRestoration = 'manual';
  // Use instant where supported, fallback to auto
  try {
    window.scrollTo({ top: 0, left: 0, behavior: 'instant' });
  } catch(_) {
    window.scrollTo(0, 0);
  }
});

// Defensive: if a hash slipped into the URL, strip it and reset.
if (location.hash && !location.hash.startsWith('#top')) {
  const url = location.href.replace(location.hash, '');
  history.replaceState(null, '', url);
  window.scrollTo(0, 0);
}
