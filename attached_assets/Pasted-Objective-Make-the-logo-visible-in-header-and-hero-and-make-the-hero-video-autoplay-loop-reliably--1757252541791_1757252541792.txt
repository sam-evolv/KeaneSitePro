Transcode the video for web compatibility
Some encoders output HEVC/odd color profiles that iOS/Chrome won’t play inline. Transcode to H.264 + faststart, no audio, 1080p, yuv420p:
ffmpeg -y -i public/assets/hero/hero-loop-src.mp4 \
  -vf "scale=1920:-2" -an -c:v libx264 -profile:v high -pix_fmt yuv420p \
  -movflags +faststart -crf 20 -preset veryfast \
  public/assets/hero/hero-loop.mp4
Generate a poster frame:
ffmpeg -y -ss 00:00:02 -i public/assets/hero/hero-loop.mp4 \

JS toggle + reduced motion
In public/app.js:
(function(){
  const v = document.getElementById('heroVideo');
  const b = document.getElementById('toggleVideo');
  if(!v || !b) return;

  const prefersReduced = matchMedia('(prefers-reduced-motion: reduce)');
  const apply = () => {
    if(prefersReduced.matches){ v.pause(); v.removeAttribute('autoplay'); b.textContent='Play Background'; }
  };
  apply(); prefersReduced.addEventListener('change', apply);

  b.addEventListener('click', () => {
    if(v.paused){ v.play(); b.setAttribute('aria-pressed','false'); b.textContent='Pause Background'; }
    else { v.pause(); b.setAttribute('aria-pressed','true'); b.textContent='Play Background'; }
  });
})();
